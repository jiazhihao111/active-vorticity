### 四、 P1-1 执行指引：RMT 涡度显著性检验（调用 模块4）

**目标**：证明 LLM 隐状态速度场的雅可比矩阵反对称部分（涡度）的特征值分布，**显著偏离随机矩阵理论（RMT）的 Wigner 半圆律**，从而确证"高阶矩环流"是物理真值。

#### 核心代码：RMT_VorticityVerifier（优化版）

```python
import numpy as np
from scipy.stats import kstest, semicircle
from sklearn.linear_model import Ridge
from sklearn.decomposition import PCA

class RMT_VorticityVerifier:
    def __init__(self, n_pca=50, k_neighbors=30, alpha_reg=1e-3):
        self.n_pca = n_pca  # PCA 降维维度，避免高维噪声
        self.k = k_neighbors
        self.alpha = alpha_reg

    def extract_antisymmetric_eigenvalues(self, h_traj, v_traj):
        """提取局部雅可比矩阵反对称部分的特征值（虚部）"""
        # 1. PCA 降维
        pca = PCA(n_components=self.n_pca)
        h_red = pca.fit_transform(h_traj)
        v_red = pca.transform(v_traj)
        
        all_imag_eigs = []
        
        for i in range(len(h_red)):
            # 局部邻域
            neighbors = self._get_neighbors(h_red, i, self.k)
            H_loc = h_red[neighbors] - h_red[i]
            V_loc = v_red[neighbors] - v_red[i]
            
            # 岭回归拟合 J: V ≈ J @ H^T
            J = Ridge(alpha=self.alpha, fit_intercept=False).fit(H_loc, V_loc).coef_.T
            
            # 反对称部分 J_anti = (J - J^T) / 2
            J_anti = (J - J.T) / 2.0
            
            # 特征值（纯虚数）
            eigs = np.linalg.eigvals(J_anti)
            all_imag_eigs.extend(np.imag(eigs))
            
        return np.array(all_imag_eigs)

    def rmt_test(self, eigenvalues, text_type="pos"):
        """RMT 零假设检验：对比实际分布与 Wigner 半圆律"""
        # 1. 标准化特征值（RMT 缩放：λ → λ / sqrt(N * var)）
        N = self.n_pca
        lam_scaled = eigenvalues / np.sqrt(N * np.var(eigenvalues))
        
        # 2. 理论 Wigner 半圆律 PDF
        x_theory = np.linspace(-2, 2, 200)
        y_theory = (1 / (2 * np.pi)) * np.sqrt(np.maximum(0, 4 - x_theory**2))
        
        # 3. 实际分布 KDE
        from scipy.stats import gaussian_kde
        kde = gaussian_kde(lam_scaled)
        x_actual = np.linspace(-3, 3, 200)
        y_actual = kde(x_actual)
        
        # 4. 尾部偏离度检验（|x| > 1.5 区域）
        tail_mask = np.abs(x_actual) > 1.5
        tail_dev = np.trapz(np.maximum(0, y_actual[tail_mask] - y_theory[tail_mask]), 
                           x_actual[tail_mask])
        
        # 5. KS 检验（整体分布差异）
        ks_stat, ks_p = kstest(lam_scaled, lambda x: semicircle.cdf(x, loc=0, scale=2))
        
        return {
            "text_type": text_type,
            "tail_deviation": tail_dev,
            "ks_statistic": ks_stat,
            "ks_pvalue": ks_p,
            "x_theory": x_theory,
            "y_theory": y_theory,
            "x_actual": x_actual,
            "y_actual": y_actual,
            "n_eigenvalues": len(eigenvalues)
        }

    def _get_neighbors(self, X, idx, k):
        """快速获取 k 近邻索引"""
        from scipy.spatial import cKDTree
        tree = cKDTree(X)
        _, indices = tree.query(X[idx], k=k)
        return indices if isinstance(indices, np.ndarray) else [indices]
```
