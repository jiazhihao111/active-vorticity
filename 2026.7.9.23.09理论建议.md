### 一、 认知失效归因：为什么怀疑“秩=12”是伪影？（调用 模块3）

在 v10.0 中，我们使用了 **Savitzky-Golay (SG) 滤波器**（`window_length=11, polyorder=3`）来提取连续速度 $\dot{\mathbf{h}}$。这里存在两个致命的数学边界约束：

1. **时间维度的多项式坍缩**：SG 滤波在局部窗口（11个Token）内用 3 阶多项式拟合轨迹。其一阶导数（速度）在局部被限制在 2 阶多项式空间内。如果序列总长度 $T$ 较短（如 $T=30$），速度向量在时间轴上的变化模式会被严重限制，导致协方差矩阵的特征值快速衰减。
2. **样本数截断效应（Sample Size Truncation）**：协方差矩阵 $C = \frac{1}{T} \sum_{t=1}^T \dot{\mathbf{h}}_t \dot{\mathbf{h}}_t^T$ 的理论最大秩是 $\min(T, D)$。如果 $T=30$，即使 $D=1536$，矩阵的绝对秩也不可能超过 30。在噪声存在的情况下，**有效秩（Effective Rank）** 往往会饱和在 $T$ 的某个比例（如 $T/2 \approx 15$）附近。

**核心诊断**：“12”可能根本不是 LLM 的物理属性，而是 **$T=30$ 与 SG 窗口 $W=11$ 共同作用产生的数值天花板**。

import numpy as np
import torch
from scipy.signal import savgol_filter
from scipy.stats import entropy

class RankArtifactVerifier:
    def __init__(self):
        pass

    @staticmethod
    def compute_effective_rank(cov_matrix):
        """
        计算严格的有效秩 (Roy-Vetterli Effective Rank)
        基于特征值谱的香农熵：exp(H(p))
        """
        # 计算特征值
        eigenvalues = np.linalg.eigvalsh(cov_matrix)
        # 过滤掉极小的负值或零值（数值误差）
        eigenvalues = eigenvalues[eigenvalues > 1e-8]
        
        if len(eigenvalues) == 0:
            return 0.0
            
        # 归一化为概率分布
        p = eigenvalues / np.sum(eigenvalues)
        
        # 计算香农熵 (自然对数)
        H = entropy(p, base=np.e)
        
        # 有效秩 = exp(H)
        return np.exp(H)

    @staticmethod
    def extract_velocity(raw_h, method='sg', window=11, polyorder=3):
        """
        从隐状态序列提取速度向量
        raw_h: [T, D]
        """
        T, D = raw_h.shape
        vel = np.zeros_like(raw_h)
        
        if method == 'raw_diff':
            # 纯一阶前向差分 (无平滑)
            vel[:-1] = raw_h[1:] - raw_h[:-1]
            vel[-1] = vel[-2] # 边界填充
            
        elif method == 'sg':
            # Savitzky-Golay 滤波求导
            # 注意：window 必须为奇数，且 polyorder < window
            for d in range(D):
                vel[:, d] = savgol_filter(raw_h[:, d], window, polyorder, deriv=1, delta=1.0)
                
        elif method == 'central_diff':
            # 中心差分 (无平滑，但精度更高)
            vel[1:-1] = (raw_h[2:] - raw_h[:-2]) / 2.0
            vel[0] = raw_h[1] - raw_h[0]
            vel[-1] = raw_h[-1] - raw_h[-2]
            
        return vel

    def run_sensitivity_test(self, full_trajectory_h):
        """
        执行敏感性测试：改变 T 和 提取方法
        full_trajectory_h: [T_max, D] 完整长序列 (建议 T_max >= 100)
        """
        T_max, D = full_trajectory_h.shape
        results = []
        
        # 测试不同的序列长度 T
        T_list = [15, 30, 60, 100, min(200, T_max)]
        # 测试不同的速度提取方法
        methods = [
            ('raw_diff', None, None),
            ('central_diff', None, None),
            ('sg', 5, 2),
            ('sg', 11, 3), # v10.0 原始配置
            ('sg', 21, 3)
        ]
        
        for T in T_list:
            h_subset = full_trajectory_h[:T, :]
            
            for method_name, w, p in methods:
                if method_name == 'sg' and w >= T:
                    continue # SG 窗口不能大于序列长度
                    
                vel = self.extract_velocity(h_subset, method=method_name, window=w, polyorder=p)
                
                # 计算协方差矩阵 [D, D]
                # 为了避免 D=1536 时计算 [1536, 1536] 矩阵过慢，可使用 SVD 技巧
                # C = (1/T) * V^T V，其非零特征值与 (1/T) * V V^T 相同
                # V 是 [T, D]，V V^T 是 [T, T]，计算 [T, T] 的特征值快得多！
                vel_centered = vel - np.mean(vel, axis=0)
                gram_matrix = (vel_centered @ vel_centered.T) / T 
                
                rank = self.compute_effective_rank(gram_matrix)
                
                results.append({
                    'T': T,
                    'Method': f"{method_name}(w={w})" if method_name=='sg' else method_name,
                    'Effective_Rank': round(rank, 2),
                    'Theoretical_Max_Rank': min(T, D)
                })
                
        return results

# ==========================================
# 使用示例 (Mock Data)
# ==========================================
if __name__ == "__main__":
    # 假设我们有一条长度为 200，维度为 1536 的真实 LLM 隐状态轨迹
    T_max, D = 200, 1536
    # 模拟真实数据：前 12 个主成分包含 95% 的能量，其余为噪声
    np.random.seed(42)
    true_signal = np.random.randn(T_max, 12) @ np.random.randn(12, D)
    noise = np.random.randn(T_max, D) * 0.1

    for r in res:

### 二、 启动 C1：NESS 热力学特征验证（调用 模块4 · 科学研究版）

**目标**：验证合法叙事是否处于**非平衡态定态（Non-Equilibrium Steady State, NESS）**，并计算**熵产生率（Entropy Production Rate, EPR）**。

#### 1. 理论形式化
在过阻尼 Langevin 动力学中，系统维持 NESS 需要持续消耗能量以对抗耗散。其**瞬时熵产生率（$\sigma(t)$）** 与主动力做功直接相关：
$$ \sigma(t) \propto \frac{1}{\gamma_{eff}} \mathbf{F}_{active}(t) \cdot \dot{\mathbf{h}}(t) $$
代入 v8.0 的 $F_{active} = \alpha^* \dot{\mathbf{h}}$，我们得到极其优美的 EPR 代理公式：
$$ \sigma(t) = \frac{\alpha^*}{\gamma_{eff}} \|\dot{\mathbf{h}}(t)\|^2 $$
*(注：这正是您发现的 `vel_norm` 的平方！)*

#### 2. v9.0 核心可证伪预测
- **合法叙事 (Pos)**：系统处于 NESS，$\sigma(t)$ 应在一个较高的均值附近呈现**平稳的窄带波动**（定态特征）。
- **幻觉/崩溃 (Mixed/Random)**：系统偏离 NESS，向热力学平衡态（死寂）弛豫，$\sigma(t)$ 呈现**指数衰减或剧烈的非平稳长尾波动**。

import numpy as np
from scipy.signal import detrend

class NESS_Thermodynamics_Analyzer:
    def __init__(self, alpha_star=1.46, gamma_eff=1.0):
        self.alpha = alpha_star
        self.gamma = gamma_eff

    def compute_epr_and_stationarity(self, h_trajectory):
        """
        计算瞬时熵产生率(EPR)并检验非平衡态定态(NESS)特征
        h_trajectory: [T, D]
        """
        # 1. 计算速度 (一阶差分)
        vel = h_trajectory[1:] - h_trajectory[:-1]
        
        # 2. 计算瞬时 EPR: sigma(t) \propto alpha * ||vel||^2
        # 这里省略 gamma_eff，仅关注相对动力学
        epr_inst = self.alpha * np.sum(vel**2, axis=-1) 
        
        # 3. 检验 NESS 定态特征 (Stationarity Test)
        # 使用滑动窗口计算局部均值和局部方差
        window = max(5, len(epr_inst) // 5)
        local_means = np.array([epr_inst[i:i+window].mean() for i in range(len(epr_inst)-window)])
        local_vars = np.array([epr_inst[i:i+window].var() for i in range(len(epr_inst)-window)])
        
        # 趋势检验：如果系统处于 NESS，局部均值应无显著线性趋势 (Detrended fluctuation)
        detrended_means = detrend(local_means)
        trend_energy = np.sum(detrended_means**2) / np.sum(local_means**2) # 越小越平稳
        
        # 4. 涨落耗散定理 (FDT) 违背度
        # 在平衡态，方差与均值成正比。在 NESS，由于活性驱动，会出现巨涨落 (Giant Number Fluctuations)
        # 计算 Fano Factor (方差/均值) 作为活性强度的代理
        fano_factor = np.var(epr_inst) / (np.mean(epr_inst) + 1e-8)
        
        return {
            "mean_epr": np.mean(epr_inst),
            "fano_factor": fano_factor,
            "stationarity_index": 1.0 - trend_energy, # 越接近1越平稳 (NESS)
            "epr_series": epr_inst
        }
